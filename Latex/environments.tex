\newtheorem{THM}{Thm}[section]

\theoremstyle{definition}
\newtheorem{DEFN}[THM]{Def}
\newtheorem{BEM}[THM]{}
\newtheorem{BSP}[THM]{Bsp}

	\condclearpage\begin{mdframed}[linecolor=white,backgroundcolor=ThmColor]\begin{PROP}}{\end{PROP}\end{mdframed}}
